(function () {
    "use strict";
 
    describe("Testing the dashboard controller: ", function () {
        var controller, rootScope, SERVER_URL, state, httpBackend, stateParams;
 
        beforeEach(function () {
            module("huqasAppConfig");
            module("ui.router");
            module("huqas.common.forms");
            module("huqas.events.controllers");
            module("huqas.events.services");
 
            inject(["$rootScope", "$controller", "SERVER_URL", "$httpBackend",
                "$state", "huqas.events.wrappers", "$stateParams",
                function ($rootScope, $controller, SERVER_URL, _$httpBackend_,
                    $state, wrappers, $stateParams) {
                    rootScope = $rootScope;
                    httpBackend = _$httpBackend_;
                    SERVER_URL = SERVER_URL;
                    state = $state;
                    stateParams = $stateParams;
                    wrappers = wrappers;
                    controller = function (name, data) {
                        return $controller("huqas.events.controllers." + name, data);
                    };
                }
            ]);
        });
 
        it("should test main events controller listing events", function () {
            var scope = rootScope.$new();
            var data = {
                "$scope": scope
            };
            var httpResponse = {"name" : "get"};
            httpBackend.expectGET(SERVER_URL +
                "api/v1/events/test_events/").respond(200, httpResponse);
            controller("main", data);
            httpBackend.flush();
            httpBackend.verifyNoOutstandingRequest();
            httpBackend.verifyNoOutstandingExpectation();
        });
        it("should test main events controller listing events: fail",
            function () {
            var scope = rootScope.$new();
            var data = {
                "$scope": scope
            };
            httpBackend.expectGET(SERVER_URL +
                "api/v1/events/test_events/").respond(500, {});
            controller("main", data);
            httpBackend.flush();
            httpBackend.verifyNoOutstandingRequest();
            httpBackend.verifyNoOutstandingExpectation();
        });
        it("should test editing a test event",
            function () {
            inject(["huqas.common.forms.changes", function (formChanges) {
                var scope = rootScope.$new();
                var data = {
                    "$scope": scope,
                    "$stateParams" : {
                        event_id : 1
                    },
                    "$state" : state,
                    "huqas.common.forms.changes": formChanges
                };
                spyOn(state, "go");
                httpBackend.expectGET(SERVER_URL +
                    "api/v1/catalog/programs/").respond(200, {results : []});
                spyOn(formChanges, "whatChanged").and.returnValue({"name": "get"});
                controller("edit_event", data);
                httpBackend.expectGET(SERVER_URL +
                    "api/v1/events/test_event/1/").respond(200, {results :[]});
                httpBackend.expectPATCH(SERVER_URL +
                    "api/v1/events/test_event/1/")
                    .respond(200, {"name" :"get"});
                data.$scope.save();
                httpBackend.flush();
                httpBackend.verifyNoOutstandingRequest();
                httpBackend.verifyNoOutstandingExpectation();
            }]);
        });
        it("should test creating a test event",
            function () {
            inject(["huqas.common.forms.changes", function (formChanges) {
                var scope = rootScope.$new();
                var data = {
                    "$scope": scope,
                    "$stateParams" : {
                        event_id : null
                    },
                    "$state" : state,
                    "huqas.common.forms.changes": formChanges
                };
                spyOn(state, "go");
                httpBackend.expectGET(SERVER_URL +
                    "api/v1/catalog/programs/").respond(200, {results : []});
                spyOn(formChanges, "whatChanged").and.returnValue({"name": "get"});
                controller("edit_event", data);
                httpBackend.expectPOST(SERVER_URL +
                    "api/v1/events/test_event/")
                    .respond(200, {"name" :"get"});
                data.$scope.save();
                httpBackend.flush();
                httpBackend.verifyNoOutstandingRequest();
                httpBackend.verifyNoOutstandingExpectation();
            }]);
        });
        it("should test creating a test event",
            function () {
            inject(["huqas.common.forms.changes", function (formChanges) {
                var scope = rootScope.$new();
                var data = {
                    "$scope": scope,
                    "$stateParams" : {
                        event_id : null
                    },
                    "$state" : state,
                    "huqas.common.forms.changes": formChanges
                };
                spyOn(state, "go");
                httpBackend.expectGET(SERVER_URL +
                    "api/v1/catalog/programs/").respond(200, {results : []});
                spyOn(formChanges, "whatChanged").and.returnValue({});
                controller("edit_event", data);
                data.$scope.save();
                httpBackend.flush();
                httpBackend.verifyNoOutstandingRequest();
                httpBackend.verifyNoOutstandingExpectation();
            }]);
        });
    });
})();
